机选彩票

随机彩票

彩票规则：分篮球(6个，1~32不能重复)和红球(1个，1~16)

机选:选择出来的6个篮球按照从小到大的原则排列,红球放在最后

首先：我们来确定一下布局，设计一下我们要显示的效果，这里我们需要一个按钮来控制开始，结束我们在这里使用定时器来完成（也可以通过其他方式来实现，自己想想其他方法）

    <style>
        #hong{
            background-color: red;
            font-size: 20px;
        }
        #num{
            background-color: gray;
            font-size: 20px;
        }
    </style>
    <body>
        <button id="btn">随机一次</button>
        <div>
            <span id="num"></span>
            <span id="hong"></span>
        </div>
    </body>

定义一个按钮,用来控制开始，定义两个span标签，用来显示随机的篮球和红球的号码

下面来看一下效果，注意在span标签中，是没有任何内容的，所以，它们现在是没有高度和宽度的，应该不会显示出来

接下来：我们要开始定义变量，用来获取每个标签 (注：下面的定义都是在script标签中)

    <script>
        var nums=document.getElementById("num");
        var btn=document.getElementById("btn");
        var hongs=document.getElementById("hong");
    </script>

实现机选最基础的就是产生随机数，这是这个小项目的前提

那么，我们就来定义一个函数，产生随机数(函数名:suiji,有两个参数(min,max),调用的时候不要忘记传递两个参数,与之相对应)

    function suiji(min,max){
        return Math.round(Math.random()*(max-min)+min);
    }

产生随机数的函数定义完成，我们还要给产生的随机数进行一个从小到大的排列，so，现在我们需要一个函数来完成这个功能

    //对数组进行一个从小到大的排序
    function sortNumber(a,b){
        return a-b;
    }

接下来就是重点了:

我们要定义一个函数，来判断产生的随机数是不是有重复，重复的直接去除，不重复的存储下来

    function select(){
        var ch=[];
        while(ch.length<6){
            var num=suiji(0,32);        //在0-32中产生随机数
            for(var i=0;i<ch.length;i++){
                if(num==ch[i]){
                    break;    //跳出for循环
                }
            }
            if(i==ch.length){
                ch.push(num);
            }
        }
        ch=ch.sort(sortNumber);    //调用sortNumber时，不需要加参数，函数名后面不用写小括号
        return ch;
    }

我们来看一下这段代码，首先定义一个select()的函数，主要功能就是产生一组(6个)随机数，并判断是否重复，不重复的保存，最后将产生的数组元素进行返回

我们还有红球的随机数需要产生，我们现在来定义函数

    function hong(){
        var one=suiji(0,16);    //同样，使用之前定义的产生随机数的函数，这里只有一个数值需要产生，所以不需要进行重复数值的检查
        return one;
    }

记住，我们还有一个按钮来控制开始，定时用来结束，下面我们来写一下代码:

    btn.onclick=function(){        //添加点击事件
        //定义循环定时器,每次循环间隔100ms
        var xunhuan=setInterval(function(){
            nums.innerHTML=select();
            hongs.innerHTML=hong();
        },100)
        //清除循环定时器,延迟等待的时间为1000ms
        setTimeout(function(){
            clearInterval(xunhuan);
        },1000);
    }

给按钮添加一个点击事件，在事件中设置了一个循环定时器，每隔100ms执行一次，同样，还设置了一个清除循环定时器的函数，时间设置为1000ms，也就是说，当按下按钮之后，等待1000ms，循环定时器停止，程序执行结束

好了，现在我们要的所有功能就已经全部定义完成了，我们来整理一下代码，如下:

<!doctype html>
<html>
    <head>
        <meta charset="utf-8"/>
        <title>机选彩票</title>
        <style type="text/css">
            #hong{
                background-color: red;
                font-size: 20px;
            }
            #num{
                background-color: gray;
                font-size: 20px;
            }
        </style>
    </head>
    <body>
        <button id="btn">随机一次</button>
        <div>
            <span id="num"></span>
            <span id="hong"></span>
        </div>
    </body>
    <script type="text/javascript">
        var nums=document.getElementById("num");
        var btn=document.getElementById("btn");
        var hongs=document.getElementById("hong");

        btn.onclick=function(){
            var xunhuan=setInterval(function(){
                nums.innerHTML=select();
                hongs.innerHTML=hong();
            },100)
            setTimeout(function(){
                clearInterval(xunhuan);
            },1000);
        }

        function hong(){
            var one=suiji(0,16);
            return one;
        }

        function select(){
            var ch=[];
            while(ch.length<6){
                var num=suiji(0,32);        //在0-32中产生随机数
                for(var i=0;i<ch.length;i++){
                    if(num==ch[i]){
                        break;    //跳出for循环
                    }
                }
                if(i==ch.length){
                    ch.push(num);
                }
            }
            ch=ch.sort(sortNumber);    //调用sortNumber时，不需要加参数，函数名后面不用写小括号
            return ch;
        }

        //随机数按照从小到大的顺序排列
        function sortNumber(a,b){
            return a-b;
        }

        //确定随机数的范围
        function suiji(min,max){
            return Math.round(Math.random()*(max-min)+min);
        }
    </script>
</html>

可以尝试着敲一遍代码，看看效果!
