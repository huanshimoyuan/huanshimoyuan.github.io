仿写JQery之选择器

选择器

在使用JQuery时,$()里面接收参数，通过传递的参数，来查找我们的目标对象或者执行某种方法

传递的参数有三种可能：1.接收的可能是个方法(e.g:function(){}) 2.可能是字符串(e.g:“div .one”) 3.也可能是对象(原生)(e.g:“document.getElementById(“two”)”)

使用$()时，实际上是实例化出来一个对象，那么我们先来定义一个方法:

    function $(obj){
        return new Jq(obj);
    }

想要实例化一个对象，我们就需要构造函数，在上面，我实际上已经使用了，现在就来完成这个构造函数:

    function Jq(obj){                //这里的函数名第一个字母大写，是为了与普通函数进行区分，增强代码的可读性
        this.elements = [];            //给对象添加属性，用来存储，我们接下来找到的目标对象
        if(typeof(obj)=="function"){        //判断参数是哪种类型
            ready(obj);                            //这里我先调用了一个函数，这个函数使用了添加监听事件(添加事件需要考虑IE的兼容，所以我们需要使用这个辅助函数)和执行接收的方法,需要注意一点，参数是方法的时候，是在HTML、CSS、JS文件加载完成之后执行的
        }else if(typeof(obj)=="string"){        //匹配目标对象，单独封装函数，函数有返回值，进行存储
            this.elements = getElement(obj);
        }else{                                    //参数是对象,将对象添加到目标对象数组中
            this.elements.push(obj);
        }
    }

上面，调用了两个辅助函数，接下来，我们来完成函数的功能

    //参数是方法时，需要执行这个方法，同时，我们还要解决兼容问题
    function ready(fn){        //当加载完成时，执行传递的方法
        //首先，来写标准浏览器的方法
        if(window.addEventListener){        //DOMContentLoaded用来监听加载状态的
            document.addEventListener("DOMContentLoaded",fn,false);    //直接使用DOM2级添加事件的方法 
        }else{        //IE浏览器
            var createS = document.createElement("script");
            createS.defer = true;    //规定是否对脚本执行进行延迟，直到页面加载完成为止,只有IE浏览器有这个属性
            var headS = document.querySelector("head");
            headS.appendChild(createS);
            createS.onreadystatechange = function(){
                if(createS.readyState == "complete"){
                    fn();
                }
            }
        }
    }

    //参数是字符串时，我们需要对字符串进行处理，从而获得我们想要获取的对象
    function getElement(str){
        //在书写是，有可能不会严格遵守规范来写，例如:$(" div  .one  p ")，这里就产生了很多空格
        //我们需要先把格式修改为符合规范的，再去查找目标对象，使用正则表达式，进行操作
        var regStandard1 = /^\s+|\s+$/g;
        str = str.replace(regStandard1,"");    //将开头或结尾的空格删除
        var regStandard2 = /\s+/g;
        str = str.replace(regStandard2," ");    //将所有的空格用一个空格代替,得到标准格式$("div .one p")
        var tar = str.split(" ");    //将字符串分割为数组,遍历数组中的元素，查找目标对象
        var result = [];    //存储找到的目标对象
        var rangen = [document];    //查找范围，第一次是在document中查找，下次就是在第一次查找的结果中查找，所以，我们需要变更查找范围
        for(var i=0;i<tar.length;i++){
            result = [];    //清空之前的结果
            for(var j=0;j<rangen.length;j++){
                //判断查找的目标是什么类型  是类名 标签名 ID名
                if(/^\./.test(tar[i])){
                    var className = tar[i].substring(1);   //删除开头的.在进行对比查找
                    var classElement = findClass(rangen[j],className);    //调用findClass方法查找，考虑兼容IE浏览器
                    for(var k=0;k<classElement.length;k++){
                        result.push(classElement[k]);    //保存查找到的每个目标
                    } 
                }else if(/^#/.test(tar[i])){
                    var idName = tar[i].substring(1);   //删除开头的#号
                    var idElement = document.getElementById(idName);    //只有document有getElementById方法
                    result.push(idName);
                }else{
                    var tagName = rangen[j].getElementsByTagName(tar[i]);
                    for(var k=0;k<tagName.length;k++){
                        result.push(tagName[k]);
                    }
                }
            }
            rangen = result;    //查找到的结果，作为下次查找的范围
        }
        return result;        //返回查找到的目标对象
    }

    //查找类名的方法
    function findClass(obj,className){
        if(obj.getElementsByClassName){        //标准浏览器
            return obj.getElementsByClassName(className);
        }else{        //IE浏览器
            var allNode = obj.getElementsByTagName("*");    //获取obj下所有的标签名
            var rightNode = [];        //存储符合条件的结果
            var regRight = new RegExp("\\b"+className+"\\b");  //匹配类名的正则表达式
            for(var i=0;i<allNode.length;i++){
                if(regRight.test(allNode[i].className)){
                    rightNode.push(allNode[i]);        //存储结果
                }
            }
            return rightNode;        //返回最终的结果
        }
    }

到这里,我们的选择器就完成了,可以去测试一下: console.log($(“ .div1 #one p “));

当然，这只是初步的选择器,有些地方还需要进行优化和bug的修复
